CREATE TABLE Asignacion.asignacion_envio (
	id INT auto_increment NOT NULL,
	fechaAsignacion DATE NULL,
	idChofer INT NULL,
	idVehiculo INT NULL,
	idEnvio INT NULL,
	estado varchar(5) NULL,
	creado_por INT NULL,
	creado_en DATE NULL,
	actualizado_por INT NULL,
	actualizado_en DATE NULL,
	CONSTRAINT asignacion_envio_pk PRIMARY KEY (id)
)
ENGINE=InnoDB
DEFAULT CHARSET=utf8mb4
COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE Asignacion.vehiculo (
	id INT NULL,
	placa varchar(20) NULL,
	marca varchar(50) NULL,
	modelo varchar(50) NULL,
	capacidadCarga DECIMAL (10,2) NULL,
	estado varchar(5) NULL,
	creado_por INT NULL,
	creado_en DATE NULL,
	actualizado_por INT NULL,
	actualizado_en DATE NULL
)
ENGINE=InnoDB
DEFAULT CHARSET=utf8mb4
COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE Asignacion.mantenimiento (
	id INT auto_increment NOT NULL,
	descripcion varchar(500) NULL,
	estado varchar(5) NULL,
	fechaInicio DATE NULL,
	fechaFin DATE NULL,
	idVehiculo INT NULL,
	creado_por INT NULL,
	creado_en DATE NULL,
	actualizado_por INT NULL,
	actualizado_en DATE NULL
)
ENGINE=InnoDB
DEFAULT CHARSET=utf8mb4
COLLATE=utf8mb4_0900_ai_ci;
